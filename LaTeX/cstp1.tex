\subsection{Pseudo code du tri "Bubble sort"} 

\begin{lstlisting}[language=C, caption=Pseudo code du tri "Bubble sort"]
    input: int N, int D
    output: int
    begin
      res $\gets$ 0
      while N $\geq$ D 
        N $\gets$ N - D
        res $\gets$ res + 1      
      end
      return res
    end     
\end{lstlisting}
